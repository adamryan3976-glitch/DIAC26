/** * DIAC 2026 - Staff Hub Backend * Handles Login, Student Data, and Schedule */ function doGet(e) { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. LOGIN ACTION if (action === "login") { const user = e.parameter.user ? e.parameter.user.toLowerCase().trim() : ""; const pass = e.parameter.pass ? e.parameter.pass.trim() : ""; const sheet = ss.getSheetByName("App Staff List"); const data = sheet.getDataRange().getValues(); // Column Index Mapping for App Staff List: // A: First, B: Last, C: Username, D: Password, E: isHoc for (let i = 1; i < data.length; i++) { const dbUser = String(data[i][2]).toLowerCase().trim(); const dbPass = String(data[i][3]).trim(); const isHoc = !!data[i][4]; // Boolean check for Column E if (dbUser === user && dbPass === pass) { return ContentService.createTextOutput(JSON.stringify({ success: true, isHoc: isHoc })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ success: false, error: "Invalid Credentials" })).setMimeType(ContentService.MimeType.JSON); } // 2. STUDENTS ACTION if (action === "students") { const sheet = ss.getSheetByName("Student App Data"); const data = sheet.getDataRange().getValues(); const results = []; // Starting at i=1 to skip the header row for (let i = 1; i < data.length; i++) { const row = data[i]; // Object key mapping (a=A, b=B, etc.) for index.html use results.push({ a: row[0], // First Name b: row[1], // Last Name c: row[2], // Student Number g: row[6], // Cabin n: row[13], // School Planning to Attend o: row[14], // Photo URL p: row[15], // Dietary q: row[16], // Allergies ac: row[28] // Preferred Full Name (Column AC) }); } return ContentService.createTextOutput(JSON.stringify(results)) .setMimeType(ContentService.MimeType.JSON); } // 3. SCHEDULE ACTION if (action === "schedule") { const sheet = ss.getSheetByName("App Classes"); if (!sheet) return ContentService.createTextOutput("[]").setMimeType(ContentService.MimeType.JSON); const data = sheet.getDataRange().getValues(); const schedule = []; // Assume S1, S2, S3, S4 are columns. // This is a simple placeholder - adjust if your schedule sheet has time/location columns for (let i = 1; i < data.length; i++) { if (data[i][0]) { schedule.push({ time: "Session " + i, event: data[i][0], // Pulling from S1 for now location: "See Program" }); } } return ContentService.createTextOutput(JSON.stringify(schedule)) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput("Invalid Action").setMimeType(ContentService.MimeType.TEXT); }
